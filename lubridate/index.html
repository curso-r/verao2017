Exemplos com datas e horas

lakers %>% 
  select(date, time) %>% 
  mutate(`paste(date, time) %>% ymd` = paste(date, time) %>% ymd,
         `paste(date, time) %>% ymd_hm` = paste(date, time) %>% ymd_hm,
         `time %>% hm` = time %>% hm,
         `time %>% hms` = time %>% hms
         ) %>% 
  head %>%
  knitr::kable()
