media <- function(x){
  i <- 1
  tamanho <- length(x)
  soma <- 0
  for(i in 1:tamanho){
    soma <- soma + x[i]
  }
  return(soma/tamanho)
}

soma_dados <- function(n){
  soma <- 0
  for(i in 1:n){
    soma <- soma + sample(1:6, 1)
  }
  return(soma)
}
resultados <- integer(length = 1000)
for(i in 1:1000){
  resultados[i] <- soma_dados(3)
}
